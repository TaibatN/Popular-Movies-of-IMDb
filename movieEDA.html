plt.style.use('seaborn-darkgrid')
# CREATE DATA PLOT
fig, ax = plt.subplots(figsize = (20, 10))

# PLOT THE DATA
top_20 = sns.barplot(x = top_20_movies.title,
                     y = top_20_movies.vote_count,
                    palette = ('husl'))
#SET CHART AXIS
sns.set_context('poster')
plt.setp(ax.get_xticklabels(), fontsize = 20, rotation = 'vertical')

# CUSTOMIZE THE DATA LABELS
ax.set(title = "Top 20 Popular Movies",
      xlabel = 'Movie Titles',
      ylabel = 'Vote Count');
